# size = n, p_hat = 0.5, p  95 %  (alpha = 0.05) 
p_h <- 0.5  ; alpha <- 0.05
# Z = (p_hat-p_0)/sqrt(p_0 * (1-p_0)/n) 
# P-value (CI) = 0.95
# qnorm (0.025), qnorm (0.975) ->  -1.96 < Z < 1.96
# n   ,  p_0     p_0 = p_hat    

#  :  p_h + c.l * sqrt(p_h*(1-p_h)/n) < p_0 < p_h + c.u * sqrt(p_h*(1-p_h)/n) 
#  n = 30   
n <- 30
p_l <-  0.5 - 1.96 * sqrt(0.5*0.5/n)
p_u <-  0.5 + 1.96 * sqrt(0.5*0.5/n)

# 
cat(paste("   n = 30,   p_hat = 0.5  ,  p   95%  \n",
      round(p_l , 3),  "< p_0 <",round( p_u, 3) ,""))
##    n = 30,   p_hat = 0.5  ,  p   95%  
##  0.321 < p_0 < 0.679 
# 
z <- seq(-3, 3, by=0.001)
z.p <- dnorm(z)

plot(z, z.p, axes=F, type="l", 
     main="n=30, p_hat=0.5,   95%", ylab="", ylim=c(-0.04, 0.4))
lines(c(-3, 3), c(0, 0))

points(-1.96, -0.02, pch=17, col="red")
text(-1.96, -0.035, round(p_l, 3), col="red")
points(1.96, -0.02, pch=17, col="red")
text(1.96, -0.035, round(p_u, 3), col="red")

s <- seq(-1.96, 1.96, by=0.001)
s.z <- dnorm(s, mean=0, sd=1)
s <- c(-1.96, s, 1.96)
s.z <- c(0, s.z, 0)

polygon(s, s.z, density=10, col="red")
